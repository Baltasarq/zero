31 // ----------------------------- Referencia::ListaIdentificadores::calculaIds()
32 inline
33 void Referencia::ListaIdentificadores::calculaIds(const std::string & ref)
34 {
35  char * ptr;
36  const char * org;
37  unsigned int tamRef = ref.size();
38  unsigned int posIds[ tamRef + 1 ];
39 
40  init();
41  lista = new(std::nothrow) char[ tamRef + 1 ];
42 
43  if ( lista != NULL ) {
44  // Copiar la cadena, hallando el num de ids.
45  num = 1;
46  ptr = lista;
47  org = ref.c_str();
48  posIds[ 0 ] = 0;
49  while( ( *ptr = *org ) != '\0' ) {
50  if ( *ptr == CHR_SEP_IDENTIFICADOR ) {
51  // Guardar la posición del id
52  posIds[ num++ ] = ( ( ptr + 1 ) - lista );
53 
54  // Cambiar el punto por un 0
55  *ptr = 0;
56  }
57  ++ptr, ++org;
58  }
59 
60  // Reservar la memoria necesaria
61  reserva( alinea( num ) );
62 
63  // Establecer los id's
64  for(unsigned int i = 0; i < num; ++i) {
65  // Guardar índice de id en vector de posiciones
66  ids[ i ] = lista + posIds[ i ];
67  }
68 
69  // Si la referencia es '.'
70  if ( tamRef == 1
71  && *lista == CHR_SEP_IDENTIFICADOR )
72  {
73  num = 1;
74  }
75  }
76 }

317 Referencia *
318 Referencia::buscaReferencia(
319  const ListaIdentificadores &lIds,
320  Objeto *od,
321  Metodo *md)
331 {
332  Referencia *toret = NULL;
333  Objeto *obj = NULL;
334  unsigned int idN = 0;
335  unsigned int numIds = lIds.getNumero();
336 
337  // Buscar el primer objeto
338  // Si la búsqueda es global, entonces empezar en el contenedor raíz
339  // La búsqueda es global cuando el identificador empieza por '.' :
340  // ".IntStdLib.System"
341  if ( *( lIds.getId( 0 ) ) == '\0' )
342  {
343  ++idN;
344 
345  // Es la referencia a '.', el contenedor root ?
346  if ( numIds == 1 )
347  {
348  ++idN;
349  toret = Runtime::rt()->getRefAContainerRaiz();
350  }
351  else {
352  toret = buscaObjetoEnContainerActualYLib(
353  Runtime::rt()->getContainerRaiz(),
354  lIds.getId( 1 ) )
355  ;
356  }
357  }
358  else {
359  // ¿Es atributo?
360  if ( md == NULL )
361  toret = buscaObjetoEnContainerActualYLib(
362  od->getPerteneceA(), lIds.getId( 0 ) )
363  ;
364  else toret = hazSwizzlingDirecto( od, md, lIds.getId( idN ) );
365  }
366 
367  if ( toret != NULL )
368  {
369  // Buscar el siguiente ID (atributo)
370  if ( idN < numIds ) {
371  obj = toret->getObjeto();
372  ++idN;
373  }
374 
375  while( idN < numIds )
376  {
377  toret = buscaAtributo( obj, lIds.getId( idN ) );
378 
379  if ( toret != NULL ) {
380  if ( Atributo::chkVisible( (Atributo *) toret, od ) ) {
381  obj = toret->getObjeto();
382  }
383  else {
384  obj = Runtime::objetoNulo;
385  Runtime::ponExcepcion(
386  Runtime::objetoEPrivado,
387  EXC_NOVISIBLE + toret->getNombre() + '\''
388  );
389  }
390  }
391  else {
392  toret = NULL;
393  Runtime::ponExcepcionObjeto( lIds.getId( idN ) );
394 
395  break;
396  }
397 
398  ++idN;
399  }
400  } else {
401  toret = NULL;
402  Runtime::ponExcepcionObjeto( lIds.getId( idN ) );
403  }
404 
405  return toret;
406 }

409 void Referencia::hazSwizzling()
410 {
411  // Hay algo que hacer ?
412  if ( swizzled() ) {
413  return;
414  }
415 
416  Referencia *ref;
417 
418  // En qué objeto está esta referencia ?
419  Objeto *od = getPerteneceAlObjeto();
420  Metodo *md = getPerteneceAlMetodo();
421 
422  if ( lIds.getNumero() == 0 ) {
423  // Crear los identificadores de la referencia
424  lIds.calculaIds( objetoNoSwizz );
425  }
426 
427  // Buscar la referencia en cuestión
428  ref = buscaReferencia( lIds, od, md );
429 
430  // ¿Encontrada?
431  if ( ref != NULL )
432  dir = ref->getObjeto();
433  else {
434  dir = Runtime::objetoNulo;
435  Runtime::ponExcepcionObjeto( getNombreObjetoReferenciado() );
436  }
437 
438  return;
