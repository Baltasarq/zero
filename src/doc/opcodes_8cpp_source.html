361  Runtime::ponExcepcionMetodo(

375 // -------------------------------------------------------------- OcMSG::ejecuta
376 void OcMSG::ejecuta()
377 {
378  // Preparar las referencias
379  huboExcepcion = false;
380  preparaReferencias();
381 
382  // Ejecución
383  if ( !huboExcepcion ) {
384  // Comprobar argumentos
385  if ( args.size() != metDest->argumentosFormales.size() ) {
386  Runtime::ponExcepcion(
387  Runtime::objetoExcepcion, EXC_ARGNUMBER
388  + metDest->getPerteneceA()->getNombre()
389  + '.' + metDest->getNombre() + '(' + ')'
390  );
391  }
392  else
393  if ( !Metodo::chkVisible( metDest, obj ) ) {
394  Runtime::ponExcepcion(
395  Runtime::objetoEPrivado,
396  EXC_NOVISIBLE + metDest->getNombre() + '\''
397  );
398  } else {
399 
400  if ( metDest->esRecursivo() ) {
401  printf( "***Start EXE mth recursivo, Mth es: '%s'\n", metDest->getNombre().c_str() );
402  }
403  // Ejecutar el mth
404  Objeto * toret = metDest->ejecuta( &args, obj );
405 
406  if ( metDest->esRecursivo() ) {
407  printf( "***End EXE mth recursivo, Mth es: '%s'\n", metDest->getNombre().c_str() );
408  }
409 
410  // Llamar al mth de herencia dinámica
411  if ( obj->getMetHerenciaDinamica() != NULL ) {
412  obj->procHerenciaDinamica();
413  }
414 
415  // Si es recursivo, no se necesita más
416  if ( metDest->esRecursivo() ) {
417  met = metDest->getMetodoAnterior();
418 
419  if ( met == NULL ) {
420  printf( "\n*** ERROR GORDO ***, el metodo anterior es: NULL\n" );
421  printf( "Mth es: '%s'\n", metDest->getNombre().c_str() );
422  fflush( stdout );
423  } else {
424  printf( "\n*** Eliminando mth recursivo *** mth rec: '%s' mth anterior es '%s'\n",
425  metDest->getNombre().c_str(), met->getNombre().c_str());
426  }
427 
428  // Almacenar resultado en acc de llamador
429  met->getLocAcc()->ponReferencia( toret );
430 
431  // Indicar fin de ejecución
432  metDest->finEjecucion();
433 
434  metDest->descuentaRecursivo();
435  metDest->deSincroniza( met );
436  metDest->getPerteneceA()->lMets.borra( metDest );
437  } else {
438  // Almacenar resultado en acc de llamador
439  met->getLocAcc()->ponReferencia( toret );
440 
441  // Indicar fin de ejecución
442  metDest->finEjecucion();
443  }
444 
445  }
446  }
