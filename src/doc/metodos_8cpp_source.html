239 // -------------------------------------------------- Metodo::masInstrucciones()
240 void Metodo::masInstrucciones(Compilador::ListaMnemos &mnemos)
241 {
242  // Preparar el nuevo cuerpo del método, si es necesario
243  if ( instr == NULL
244  || getNumInstrucciones() < mnemos.size() )
245  {
246  inicializar();
247 
248  // Es necesario meter un NULL al final, para saber donde termina
249  instr = new(std::nothrow) Opcode *[ mnemos.size() +1 ];
250 
251  if ( instr == NULL ) {
252  throw ENoHayMemoria( "creando cuerpo del met." );
253  }
254  } else {
255  inicializarEstado();
256  }
257 
258  // Ajustar límites
259  const unsigned int numMnemos = mnemos.size();
260  unsigned int numOpsCuerpoMetodo = numMnemos;
261  finMet = instr + numOpsCuerpoMetodo;
262 
263  try {
264  // Cargar los opcodes
265  IterMet tope = instr;
266  for(register unsigned int i = 0; i < numMnemos; ++i) {
267 
268  // Es un META? (describe propiedades del método)
269  if ( dynamic_cast<NMMta *>( mnemos[i] ) != NULL ) {
270  if ( ((NMMta *) mnemos[i])->getPragma() == NMMta::Objeto ) {
271  // Es una propiedad a insertar
272  masPropiedades( (NMMta *) mnemos[i] );
273  } else {
274  procPragma( (NMMta *) mnemos[i], i );
275  }
276 
277  --numOpsCuerpoMetodo;
278  --finMet;
279  delete mnemos[i];
280  mnemos[i] = NULL;
281  continue;
282  }
283 
284  // Crear el Opcode correspondiente
285  Opcode *op = Opcode::creaOpcodeParaMnemo( mnemos[i], this );
286 
287  // Liberar el mnemo
288  delete mnemos[i];
289 
290  // Anadir la instruccion
291  *( tope ++ ) = op;
292 
293  // Si no es un DEF, y es el primero, es el fin de los DEF's
294  if ( despuesLiterales == -1
295  && ( dynamic_cast<OcDEF *>( op ) == NULL ) )
296  {
297  despuesLiterales = i;
298  }
299 
300  // Si es un RET, despues irá la seccion de excepciones
301  if ( ( dynamic_cast<OcRET *>( op ) != NULL )
302  && excep == NULL )
303  {
304  excep = tope;
305  }
306  }
307  } catch(const Excepcion &)
308  {
309  delete[] instr;
310  inicializar();
311  throw;
312  }
313 
314  // Comprobar
315  *finMet = NULL;
316  chk();
317 }

319 // --------------------------------------------------------------- Metodo::chk()
320 void Metodo::chk()
333 {
334  // Hacer los ajustes
335  IterMet inicio = instr;
336  IterMet final = excep;
337  IterMet nuevoIP;
338  IterMet it;
339 
340  for(it = inicio; it != finMet; ++it)
341  {
342  // En qué parte del cuerpo estamos ?
343  if ( it == final )
344  {
345  inicio = excep;
346  final = finMet;
347  }
348 
349  // Esté donde esté, ¿está dentro?
350  if ( dynamic_cast<OcFlw *>( *it ) != NULL )
351  {
352  nuevoIP = it + ((OcFlw *) *it)->getMnemo()->getNumero();
353 
354  if ( nuevoIP < inicio
355  || nuevoIP >= final
356  || nuevoIP == it )
357  {
358  // Cosa mala ...
359  throw EChkSaltoInvalido(
360  (*it)->getMnemo()->listar( false ).c_str()
361  );
362  }
363  else ((OcFlw*) (*it))->ponDirDirecta( nuevoIP - 1 );
364  }
365  }
366 
367  notificar( Observador::CambioEnElemento );
368 }

583 Objeto * Metodo::ejecuta(Argumentos * args, Objeto * rthis)
584 {
585  Objeto *toret;
586 
587  // Actualizar el stack
588  Runtime::gestorStack->call( rthis, this );
589 
590  // Si se está ejecutando ya, es una llamada recursiva (imposible).
591  if ( ejecutando() ) {
592  Runtime::ponExcepcion( Runtime::objetoExcepcion, EXC_REC );
593  toret = NULL;
594  goto FIN;
595  }
596 
597  _enExcepcion = false;
598 
599  // Asegurarse de que todos los opcodes van a actualizar sus referencias
600  deSincroniza();
601 
602  // Preparar final(seccPpal o seccExcep)
603  if ( excep == NULL ) {
604  excep = finMet;
605  }
606  final = excep;
607 
608  // Preparar las variables locales (cuando sea necesario)
609  preparaEstadoMetodo( rthis );
610 
611  // Comienzo de ejecucion
612  ip = comienzo;
613 
614  // Preparar los argumentos
615  if ( !convertirArgumentosAFormales( args ) ) {
616  Runtime::ponExcepcion( Runtime::objetoExcepcion, EXC_ARGNUMBER );
617  }
618 
619  // Ejecutar
620  #ifdef _UI_DEBUG
621  iuComienzaMetodo( rthis, this );
622  #endif
623 
624  for(; ip != final; ++ip)
625  {
626  // Avisar a la interfaz
627  #ifdef _UI_DEBUG
628  iuPasoMetodo( rthis, this, ip - instr );
629  #endif
630 
631  // Ejecutar la instrucción
632  ( *ip )->ejecuta();
633  }
634 
635  // Resultado de la ejecución
636  toret = reg[ NombreRegistro::__RR ]->getObjeto();
637

640  #endif
641
