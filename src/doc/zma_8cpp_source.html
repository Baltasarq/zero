952 // -------------------------------------------- MacroEnsamZ::compilaExpresion()
953 void MacroEnsamZ::compilaExpresion(const std::string &tok,
954  Metodo *& met,
955  Metodo::CuerpoMetodo &instrucc,
956  bool primerNivel)
957 /*
958  compilar una expresion de este estilo:
959  expr ::= [<ref>]+.<metodo>(<expr>)
960 
961  System.console.write(x.concat("Mundo!"))
962  A:
963  STR "Mundo!"
964  ASG __gp1
965  MSG x concat __gp1
966  MSG System.console write __acc
967 */
968 {
969  Zero::Mnemotecnico *mnemo;
970  GestorNombresLoc nl;
971 
972  lex->pasaEsp();
973 
974  if ( Zero::NombreIdentificador::compruebaId( tok )
975  || ( Zero::NombreReferencia::compruebaRef( tok )
976  && lex->getCaracterActual() != '(' ) )
977  {
978  if ( !primerNivel )
979  {
980  // Es un identificador, se asigna al acc y punto
981  mnemo = new(std::nothrow) Zero::NMSet( Zero::NombreRegistro::__ACC, tok );
982 
983  if ( mnemo != NULL )
984  instrucc.push_back( mnemo );
985  else throw Zero::ENoHayMemoria( "creando mnemo SET" );
986  } else throw Zero::ESintxIdNoValido(
987  "se esperaba asignaciÈ o mensaje"
988  );
989  }
990  else {
991  if (Zero::NombreReferencia::compruebaRef( tok ) )
992  {
993  // Es un mensaje a un objeto
994  std::string refObjeto = tok;
995  size_t posUltimoPunto = refObjeto.rfind( '.' );
996  std::string nombreMetodo = refObjeto.substr( posUltimoPunto + 1 );
997 
998  // borra '*.met'
999  refObjeto.erase(posUltimoPunto, refObjeto.size());
1000  lex->pasaEsp();
1001  if( refObjeto.empty() ) {
1002  refObjeto = ".";
1003  }
1004 
1005  // Pasar el parÈntesis, si existe
1006  if ( lex->getCaracterActual() == '(' ) {
1007  lex->avanza();
1008  }
1009  else throw Zero::ESintxIdNoValido("se esperaba apertura de parÈtesis '('");
1010 
1011  Zero::AnalizadorLexico::TipoToken tk = lex->getTipoSiguienteToken();
1012  do {
1013  // Compilar la sub-expresiÛn
1014  if (tk == Zero::AnalizadorLexico::LITNUMERICO) {
1015  Zero::Mnemotecnico * lit = parseLiteral();
1016 
1017  if ( lit == NULL ) {
1018  throw Zero::ESintxIdNoValido(
1019  "se esperaba literal"
1020  "numÈrico" );
1021  }
1022 
1023  instrucc.insert(
1024  instrucc.begin(),
1025  new Zero::NMAsg( nl.crearNuevo(),
1026  Zero::NombreRegistro::__ACC )
1027  );
1028  instrucc.insert( instrucc.begin(), lit );
1029  }
1030  else
1031  if (tk == Zero::AnalizadorLexico::LITCADENA) {
1032  Zero::Mnemotecnico * lit = parseLiteral();
1033 
1034  if (lit == NULL)
1035  throw Zero::ESintxIdNoValido("se esperaba literal cadena");
1036 
1037  // Guardar el mensaje
1038  instrucc.insert(instrucc.begin(),
1039  new Zero::NMAsg( nl.crearNuevo(),
1040  Zero::NombreRegistro::__ACC )
1041  );
1042  instrucc.insert( instrucc.begin(), lit );
1043  }
1044  else
1045  if (tk == Zero::AnalizadorLexico::IDENTIFICADOR) {
1046  // Tomar el identificador
1047  lex->getToken();
1048 
1049  // Si es un identificador, adelante
1050  if ( Zero::NombreIdentificador::compruebaId(
1051  lex->getTokenActual() )
1052  )
1053  {
1054  if ( buscaVble(
1055  objetoActual,
1056  metodoActual,
1057  lex->getTokenActual() ) )
1058  {
1059  nl.insertarNuevo(
1060  lex->getTokenActual()
1061  );
1062  } else {
1063  compilaExpresion(
1064  lex->getTokenActual(),
1065  met,
1066  instrucc
1067  );
1068 
1069  instrucc.push_back(
1070  new Zero::NMAsg(
1071  nl.crearNuevo(),
1072  Zero::NombreRegistro::__ACC)
1073  );
1074  }
1075  }
1076  else {
1077  if ( Zero::NombreReferencia::compruebaRef(
1078  lex->getTokenActual() ) )
1079  {
1080  // Es una referencia,
1081  // llamada recursiva
1082  compilaExpresion(
1083  lex->getTokenActual(),
1084  met,
1085  instrucc )
1086  ;
1087 
1088  // Pasar el ')'
1089  if ( lex->getCaracterActual() == ')' ) {
1090  lex->avanza();
1091  }
1092 
1093  instrucc.push_back(
1094  new Zero::NMAsg(
1095  nl.crearNuevo(), Zero::NombreRegistro::__ACC)
1096  );
1097  }
1098  else throw Zero::ESintxIdNoValido("se esperaba un identificador"
1099  " o referencia");
1100  }
1101  }
1102 
1103  lex->pasaEsp();
1104  tk = lex->getTipoSiguienteToken();
1105  } while(( lex->getCaracterActual() != ')')
1106  && ( !lex->esEol() ) );
1107 
1108  // Crear el mensaje
1109  Zero::NMMsg *msg = new Zero::NMMsg( refObjeto, nombreMetodo );
1110 
1111  // Argumentos
1112  while(!(nl.esVacio())) {
1113  msg->masArgumentos( nl.getSigNombre() );
1114  }
1115 
1116  instrucc.push_back( msg );
1117  }
1118  else throw Zero::ESintxIdNoValido("expresiÛn debe empezar "
1119  "por referencia o identificador")
1120  ;
1121  }
1122 }
